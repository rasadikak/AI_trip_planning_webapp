async def img_based_search(img:UploadFile= File()):
    print("hiii")
    img_bytes= await img.read()
    PIL_img= Image.open(io.BytesIO(img_bytes)).convert('RGB')
    #Image.open() expects a file-like object. BytesIO?It converts raw bytes into a fake in-memory file.
    image_input= preprocess(PIL_img).unsqueeze(0).to(device)# WE SHOULD GIVE A PIL IMAGE TO PREPROCESS 
    with torch.no_grad(): # we are not training a model, so torch.no grad, (only output, no  training the model)
        embedding= model.encode_image(image_input)  #That operation is using PyTorch tensors.
        embedding= embedding/embedding.norm(dim=-1, keepdim=True)
    embedding= embedding.cpu().numpy()
    D,I= index.search(embedding, k=3)


    results=[]
    for idx in I[0]:
        results.append({
            "label": all_labels[idx],
            "image_url": f"/dataset/{file_paths[idx]}"  # use relative path
        })
    print(results)
    return {"results":results}

```python
import pytest
from fastapi.testclient import TestClient
from backend.main import app

client = TestClient(app)

# ── Register tests ──────────────────────────────────────
def test_register_duplicate_email():
    response = client.post("/register/", data={
        "name"            : "Test User",
        "email"           : "existing@gmail.com",  # already in DB
        "password"        : "Test@1234",
        "confirm_password": "Test@1234"
    }, follow_redirects=False)
    assert response.status_code == 302
    assert "error=email_exists" in response.headers["location"]

def test_register_passwords_not_match():
    response = client.post("/register/", data={
        "name"            : "Test User",
        "email"           : "new@gmail.com",
        "password"        : "Test@1234",
        "confirm_password": "Different@1234"
    }, follow_redirects=False)
    assert response.status_code == 302
    assert "error=passwords_not_match" in response.headers["location"]

def test_register_weak_password():
    response = client.post("/register/", data={
        "name"            : "Test User",
        "email"           : "new@gmail.com",
        "password"        : "weak",
        "confirm_password": "weak"
    }, follow_redirects=False)
    assert response.status_code == 302
    assert "error=weak_password" in response.headers["location"]

# ── Login tests ──────────────────────────────────────────
def test_login_wrong_email():
    response = client.post("/login/", data={
        "username": "notexist@gmail.com",
        "password": "Test@1234"
    }, follow_redirects=False)
    assert response.status_code == 302
    assert "error=email_not_found" in response.headers["location"]

def test_login_wrong_password():
    response = client.post("/login/", data={
        "username": "existing@gmail.com",
        "password": "WrongPass@1"
    }, follow_redirects=False)
    assert response.status_code == 302
    assert "error=wrong_password" in response.headers["location"]

def test_login_not_verified():
    response = client.post("/login/", data={
        "username": "unverified@gmail.com",
        "password": "Test@1234"
    }, follow_redirects=False)
    assert response.status_code == 302
    assert "error=not_verified" in response.headers["location"]
```
